'use strict';

/**
 * @module controllers
 */

const async = require('async');
const openVeoApi = require('@openveo/api');
const errors = process.require('app/server/httpErrors.js');
const SettingsModel = process.require('app/server/models/SettingsModel.js');
const SettingsProvider = process.require('app/server/providers/SettingsProvider.js');
const WebServiceClient = process.require('/app/server/WebserviceClient.js');
const context = process.require('/app/server/context.js');

class FrontPageVideosController extends openVeoApi.controllers.Controller {

  /**
   * Defines a FrontPageVideosController to deal with promoted videos.
   *
   * @class FrontPageVideosController
   * @extends Controller
   * @constructor
   */
  constructor() {
    super();
  }

  /**
   * Handles front-page videos action to get videos ranks with details.
   *
   * Fetches configuration of front-page videos with details about the specified videos.
   * Videos fetched from OpenVeo are stored in cache.
   *
   * @method getDetailedVideosSettingsAction
   * @async
   * @param {Request} request ExpressJS HTTP Request
   * @param {Response} response ExpressJS HTTP Response
   * @param {Function} next Function to defer execution to the next registered middleware
   */
  getDetailedVideosSettingsAction(request, response, next) {
    const settingsModel = new SettingsModel(new SettingsProvider(context.database));
    const openVeoClient = WebServiceClient.get();

    /* const entity = {
      value: {
        1: 'rJPNTw4Hz',
        2: 'SJhx6vNSz'
      }
    }; */

    settingsModel.getOne('front-page-videos', null, (error, entity) => {
      if (error) {
        process.logger.error(
          error.message,
          {error: error, method: 'getDetailedVideosSettingsAction', entity: 'front-page-videos'}
        );
        next(errors.FRONT_PAGE_VIDEOS_GET_DETAILED_VIDEOS_SETTING_ERROR);
      } else if (!entity) {
        response.send({});
      } else {
        const asyncActions = [];
        const getVideoAction = (rank, videoId) => {
          return (callback) => {
            openVeoClient
            .get(`/publish/videos/${videoId}`)
            .then((response) => {
              entity.value[rank] = response.entity;
              callback();
            })
            .catch((error) => {

              // The video couldn't be fetched
              // Simply log the error
              process.logger.error(
                error.message,
                {
                  error,
                  method: 'frontPageVideosController.getDetailedVideosSettingsAction'
                }
              );

              callback();
            });
          };
        };

        for (let rank in entity.value)
          asyncActions.push(getVideoAction(rank, entity.value[rank]));

        async.parallel(asyncActions, (error, results) => {
          response.send(entity.value);
        });
      }
    });
  }

}

module.exports = FrontPageVideosController;
